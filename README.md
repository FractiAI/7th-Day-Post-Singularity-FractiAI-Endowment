    repositoryPath: './',
    branch: 'main',
    remote: 'origin',

    autoPush: true
  },

    defaultGear: AwarenessOctave.HARMONY,

    enableCloudShell: true,

    name: 'Discovery Agent',
    capabilities: []
  },

    gear: AwarenessOctave.RESONANCE,
    heroHost: 'tesla'
  }
);
